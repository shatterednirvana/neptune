9 # Exodus provides further improvements to Babel. Instead of making users tell
 
10 # us what compute, storage, and queue services they want to use (required for
 
11 # babel calls), Exodus will automatically handle this for us. Callers need
 
12 # to specify what clouds their job can run over, and Exodus will automatically
 
13 # select the best cloud for their job and run it there.
 
14 def exodus(jobs)
 
15   if jobs.class == Hash
 
16     job_given_as_hash = true
 
17     jobs = [jobs]
 
18   elsif jobs.class == Array
 
19     job_given_as_hash = false
 
20     ExodusHelper.ensure_all_jobs_are_hashes(jobs)
 
21   else
 
22     raise BadConfigurationException.new("jobs was a #{jobs.class}, which " +
 
23       "is not an acceptable class type")
 
24   end
 
25 
26   tasks = []
 
27 
28   jobs.each { |job|
 
29     ExodusHelper.ensure_all_params_are_present(job)
 
30     profiling_info = ExodusHelper.get_profiling_info(job)
 
31     clouds_to_run_task_on = ExodusHelper.get_clouds_to_run_task_on(job, 
32       profiling_info)
 
33     babel_tasks_to_run = ExodusHelper.generate_babel_tasks(job, 
34       clouds_to_run_task_on)
 
35     ExodusHelper.run_job(babel_tasks_to_run)
 
36     tasks << ExodusTaskInfo.new(babel_tasks_to_run)
 
37   }
 
38 
39   if job_given_as_hash
 
40     return tasks[0]
 
41   else
 
42     return tasks
 
43   end
 
44 end
