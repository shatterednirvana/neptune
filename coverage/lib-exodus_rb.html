15 # Exodus provides further improvements to Babel. Instead of making users tell
 
16 # us what compute, storage, and queue services they want to use (required for
 
17 # babel calls), Exodus will automatically handle this for us. Callers need
 
18 # to specify what clouds their job can run over, and Exodus will automatically
 
19 # select the best cloud for their job and run it there.
 
20 def exodus(jobs)
 
21   if jobs.class == Hash
 
22     job_given_as_hash = true
 
23     jobs = [jobs]
 
24   elsif jobs.class == Array
 
25     job_given_as_hash = false
 
26     ExodusHelper.ensure_all_jobs_are_hashes(jobs)
 
27   else
 
28     raise BadConfigurationException.new("jobs was a #{jobs.class}, which " +
 
29       "is not an acceptable class type")
 
30   end
 
31 
32   tasks = []
 
33 
34   jobs.each { |job|
 
35     ExodusHelper.ensure_all_params_are_present(job)
 
36     profiling_info = ExodusHelper.get_profiling_info(job)
 
37     clouds_to_run_task_on = ExodusHelper.get_clouds_to_run_task_on(job, 
38       profiling_info)
 
39     babel_tasks_to_run = ExodusHelper.generate_babel_tasks(job, 
40       clouds_to_run_task_on)
 
41     dispatched_tasks = ExodusHelper.run_job(babel_tasks_to_run)
 
42     tasks << ExodusTaskInfo.new(dispatched_tasks)
 
43   }
 
44 
45   if job_given_as_hash
 
46     return tasks[0]
 
47   else
 
48     return tasks
 
49   end
 
50 end

285   def self.get_minimum_val_in_data(job, profiling_info)
 
286     min_cloud = nil
 
287     min_val = 1_000_000  # infinity
 
288     optimize_for = job[:optimize_for].to_s
